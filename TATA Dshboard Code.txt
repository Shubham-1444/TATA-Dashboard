            // --- CSV Parsing Utility ---
            function parseCsv(csvText, delimiter = '\t') {
                const lines = csvText.trim().split('\n');
                if (lines.length === 0) return { headers: [], data: [] };

                const headers = lines[0].split(delimiter).map(h => h.trim());
                const data = [];
                for (let i = 1; i < lines.length; i++) {
                    const values = lines[i].split(delimiter).map(v => v.trim());
                    if (values.length === headers.length) { // Ensure row matches header count
                        let row = {};
                        headers.forEach((header, index) => {
                            row[header] = values[index];
                        });
                        data.push(row);
                    }
                }
                return { headers, data };
            }

            // --- Table Population Utility ---
            function populateTable(data) {
                const thead = dataTable.querySelector('thead');
                const tbody = dataTable.querySelector('tbody');

                // Clear existing content
                thead.innerHTML = '';
                tbody.innerHTML = '';

                if (data.headers && data.headers.length > 0) {
                    const headerRow = document.createElement('tr');
                    data.headers.forEach(headerText => {
                        const th = document.createElement('th');
                        th.textContent = headerText;
                        headerRow.appendChild(th);
                    });
                    thead.appendChild(headerRow);
                }

                if (data.data && data.data.length > 0) {
                    data.data.forEach(rowData => {
                        const tr = document.createElement('tr');
                        Object.values(rowData).forEach(cellValue => {
                            const td = document.createElement('td');
                            td.textContent = cellValue;
                            tr.appendChild(td);
                        });
                        tbody.appendChild(tr);
                    });
                } else {
                    const tr = document.createElement('tr');
                    const td = document.createElement('td');
                    td.colSpan = data.headers.length > 0 ? data.headers.length : 1;
                    td.textContent = "No data available for this file.";
                    td.style.textAlign = "center";
                    tbody.appendChild(tr);
                }
            }
